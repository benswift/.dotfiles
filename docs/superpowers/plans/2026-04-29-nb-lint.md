# nb lint Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an `nb lint` plugin that reports broken folder-prefixed `[[wikilinks]]` in any nb notebook, plus a symlink hook that auto-installs all dotfiles-tracked nb plugins.

**Architecture:** Bash plugin in `~/.dotfiles/nb/lint.nb-plugin` registers a `lint` subcommand with nb. The plugin scans every `.md` file under the target notebook, extracts folder-prefixed wikilinks (skipping fenced code blocks), and resolves each one against the filesystem (non-numeric targets) or via `nb show` (numeric targets). Output is rg-compatible by default, one-line summary with `--summary`. Tests use bats with isolated temp `NB_DIR` per test. A new section in `create_symlinks.sh` symlinks `~/.dotfiles/nb/*.nb-plugin` into `~/.nb/.plugins/`, retroactively installing the existing 4 plugins as well.

**Tech Stack:** bash, nb (via `_show`, `_notebooks`, `_subcommands` helpers it exposes to plugins), bats (run via `mise exec -- bats`), rg (already on PATH).

```bash
#!/usr/bin/env bash
#
# Shared bats helpers for nb plugin tests. Each test gets an isolated
# NB_DIR under $BATS_TEST_TMPDIR so tests don't pollute the user's
# real notebook.

# Set up an isolated nb home with the plugin under test installed.
# Creates a notebook called "test" and exports NB_DIR.
nb_test_setup() {
  export NB_DIR="${BATS_TEST_TMPDIR}/nb"
  mkdir -p "${NB_DIR}/.plugins"
  ln -s "${BATS_TEST_DIRNAME}/../nb/lint.nb-plugin" \
        "${NB_DIR}/.plugins/lint.nb-plugin"
  # nb auto-creates "home" notebook on first use. Suppress the welcome
  # banner with NB_AUTO_SYNC=0 + a discarded init call.
  NB_AUTO_SYNC=0 nb notebooks add test >/dev/null 2>&1
  NB_AUTO_SYNC=0 nb use test >/dev/null 2>&1
}

# Create a note inside the test notebook by writing a file directly to
# the notebook path. Bypasses `nb add` for speed and to control the
# filename precisely. Caller passes <relative-path> and reads content
# from stdin.
nb_test_write() {
  local _rel="$1"
  local _path="${NB_DIR}/test/${_rel}"
  mkdir -p "$(dirname "${_path}")"
  cat > "${_path}"
}

# Run `nb lint` against the test notebook with a clean PATH for nb.
# Uses BATS run so $status and $output are populated.
nb_test_lint() {
  NB_AUTO_SYNC=0 run nb test:lint "$@"
}
```

Replace the `_lint()` function body in `~/.dotfiles/nb/lint.nb-plugin`:

```bash
_lint() {
  local _summary=0
  local _notebook_arg=""
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --summary) _summary=1 ;;
      *) _notebook_arg="$1" ;;
    esac
    shift
  done

  local _notebook_path
  if [[ -n "${_notebook_arg}" ]]; then
    _notebook_path="$(_notebooks "${_notebook_arg}" --path 2>/dev/null)"
  else
    _notebook_path="$(_notebooks current --path 2>/dev/null)"
  fi
  if [[ -z "${_notebook_path}" || ! -d "${_notebook_path}" ]]; then
    printf "nb lint: notebook not found\n" >&2
    return 2
  fi

  local _broken=0
  local _broken_files=""

  # Find all .md files; skip data/icloud/, .git/, .plugins/, .archive/
  local _file
  while IFS= read -r -d '' _file; do
    local _rel="${_file#${_notebook_path}/}"
    local _line_no=0
    while IFS= read -r _line || [[ -n "${_line}" ]]; do
      _line_no=$((_line_no + 1))
      # Match every [[...]] on this line
      local _rest="${_line}"
      local _col_offset=0
      while [[ "${_rest}" =~ \[\[([^\]]+)\]\] ]]; do
        local _match="${BASH_REMATCH[1]}"
        local _full_match="[[${_match}]]"
        # Find column of the [[ in the original line
        local _idx="${_rest%%${_full_match}*}"
        local _col=$((_col_offset + ${#_idx} + 1))
        # Only check folder-prefixed targets (must contain a /)
        if [[ "${_match}" == */* ]]; then
          local _target="${_match%%#*}"  # strip anchor
          local _folder="${_target%%/*}"
          local _slug="${_target#*/}"
          local _resolved=0
          if [[ -f "${_notebook_path}/${_folder}/${_slug}.md" ]]; then
            _resolved=1
          fi
          if [[ ${_resolved} -eq 0 ]]; then
            printf "%s:%d:%d: [[%s]] -- target not found\n" \
              "${_rel}" "${_line_no}" "${_col}" "${_match}"
            _broken=$((_broken + 1))
            case " ${_broken_files} " in
              *" ${_rel} "*) ;;
              *) _broken_files="${_broken_files} ${_rel}" ;;
            esac
          fi
        fi
        # Advance past this match
        local _consumed=$((${#_idx} + ${#_full_match}))
        _col_offset=$((_col_offset + _consumed))
        _rest="${_rest:${_consumed}}"
      done
    done < "${_file}"
  done < <(find "${_notebook_path}" -type f -name '*.md' \
    -not -path "*/.git/*" \
    -not -path "*/.plugins/*" \
    -not -path "*/.archive/*" \
    -not -path "*/data/icloud/*" \
    -print0)

  if [[ ${_summary} -eq 1 ]]; then
    if [[ ${_broken} -eq 0 ]]; then
      printf "0 broken wikilinks\n"
    else
      local _file_count
      _file_count=$(printf "%s\n" ${_broken_files} | wc -l | tr -d ' ')
      printf "%d broken wikilinks across %d files\n" "${_broken}" "${_file_count}"
    fi
  fi

  if [[ ${_broken} -gt 0 ]]; then
    return 1
  fi
  return 0
}
```
